n = int(input())                            # 입력 값 선언
A = list(map(int, input().split()))       # 입력 값 만큼의 숫자 입력

d = [0]*n                                   # 연속 또는 현재 값 비교해서 저장할 배열로 처음에는 0으로 초기화
d[0] = A[0]                                 # 첫 번째 연속 값은 무조건 첫 번째 값이므로 초기화

for i in range(n):                          # 입력 값 만큼 반복
    if i == 0:                                # 첫 번째는 위에 초기화했으므로 넘김
        continue

    d[i] = max(d[i-1] + A[i], A[i])       # 점화식 : 앞에서 연속해온 값 + 현재 값과 현재 값을 비교해서 최대 값을 넣음 

print(max(d))                               # 모두 순회한 값들 중 최대 값 출력
